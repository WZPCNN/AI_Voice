```ts
  // 命令面板显示时，交给 useSlashCommands 处理
  if (handleKeyDown(event)) return true;
  // 命令面板未显示：Enter 发送
  if (event.key === 'Enter' && !event.shiftKey && !event.altKey) {
    event.preventDefault();
    handleSend();
    return true;
  }
  return false;
}
```

```ts
const handleSend = (resume = false) => {

  if (!resume && editor && !editor.isDestroyed) {
    editor.commands.clearContent();
  }
};
```

## 6. 假设与决策表

| # | 决策 | 理由 |
|---|------|------|
| A1 | uiStore 增加 `removeImage(i)` action | ChatInput 需要删除单张图片，当前 uiStore 仅有 addImage/clearImages |
| A2 | ChatPage 不再持久化到 localStorage | 后端已有 session/message 模块，useConversations 从后端加载 |
| A3 | ChatPage 移除内嵌 Sidebar | App.tsx 路由已用 Layout(Sidebar+Outlet) 包裹，子页面不应再内嵌 |
| A4 | SettingsPage 移除内嵌 Sidebar | 同 A3 |
| A5 | Layout newSession 调用 clearMessages | 新建会话时需清空 chatStore 消息，避免旧消息残留 |
| A6 | prompts 模块化为包（`prompts/`） | 支持后续扩展模式提示词，`__init__.py` 导出保持引用路径不变 |
| A7 | CI 升级到 pnpm 11 + Node 26 | 对齐实际工具链（根 package.json 声明 pnpm@11.9.0） |
| A8 | skills/mcp 仅做 UI 铺垫 | 后端实际处理逻辑需大规模改动，本次仅提供前端入口 |
| A9 | ChatPage 保留 useEditor（不抽 hook） | editor 配置含 handleKeyDown 与 handleSend 闭包，抽出会引入复杂依赖 |
| A10 | ActiveAgentsBar 内联在 ChatPage | 仅 20 行 JSX，不值得单独拆组件 |
| A11 | changesets 配置但不立即发布 | 建立版本管理基础设施，发布流程后续启用 |
| A12 | ChatInput 自包含设计（直接读 store） | 减少 props 传递，与 ModelSelector/MessageList 模式一致 |
